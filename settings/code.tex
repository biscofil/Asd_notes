	sensitive=false,
	keywords={}
	otherkeywords={<,<=,>,>=,==,=, <-,->, !=}, %operators
	keywords=[2]{for, to, downto, foreach, while, if, else, then, do, return, repeat, until, and, or, in},

	commentstyle=\color{grey},
	stringstyle=\color{red},
	numberstyle=\scriptsize

}

    mathescape=true,

}
